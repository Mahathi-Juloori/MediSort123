  <script>
    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

        // Simple triage logic based on symptoms
        setTimeout(() => {
          const primarySymptom = $('#primarySymptom').val().toLowerCase();
          const severity = parseInt($('#severity').val());
          
          let response;
          
          // Emergency symptoms
          if (['chest pain', 'shortness of breath', 'severe bleeding'].includes(primarySymptom) || severity >= 9) {
            response = {
              triage_category: "Emergency Care Needed",
              confidence: 0.90,
              recommended_action: "Seek emergency care immediately",
              probabilities: {
                "Emergency": 0.85,
                "Urgent": 0.10,
                "Non-urgent": 0.05
              }
            };
          }
          // Urgent symptoms
          else if (['fever', 'abdominal pain', 'headache'].includes(primarySymptom) || severity >= 7) {
            response = {
              triage_category: "Urgent Care Recommended",
              confidence: 0.80,
              recommended_action: "Visit urgent care within 24 hours",
              probabilities: {
                "Emergency": 0.20,
                "Urgent": 0.60,
                "Non-urgent": 0.20
              }
            };
          }
          // Non-urgent symptoms
          else {
            response = {
              triage_category: "Non-urgent Care",
              confidence: 0.75,
              recommended_action: "Schedule a doctor's appointment",
              probabilities: {
                "Emergency": 0.05,
                "Urgent": 0.20,
                "Non-urgent": 0.75
              }
            };
          }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        $('#confidenceLevel').text((data.confidence * 100).toFixed(0) + '% confident');
        $('#confidenceBar').css('width', (data.confidence * 100) + '%');
        
        let probabilitiesHtml = '<div class="mt-3"><h5>Detailed Probabilities:</h5><ul class="list-group">';
        for (const [category, prob] of Object.entries(data.probabilities)) {
          probabilitiesHtml += `
            <li class="list-group-item d-flex justify-content-between align-items-center">
              ${category}
              <span class="badge badge-primary badge-pill">${(prob * 100).toFixed(0)}%</span>
            </li>`;
        }
        probabilitiesHtml += '</ul></div>';
        $('#probabilityDetails').html(probabilitiesHtml);
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

        // Simple triage logic based on symptoms
        setTimeout(() => {
          const primarySymptom = $('#primarySymptom').val().toLowerCase();
          const severity = parseInt($('#severity').val());
          
          let response;
          
          // Emergency symptoms
          if (['chest pain', 'shortness of breath', 'severe bleeding'].includes(primarySymptom) || severity >= 9) {
            response = {
              triage_category: "Emergency Care Needed",
              confidence: 0.90,
              recommended_action: "Seek emergency care immediately",
              probabilities: {
                "Emergency": 0.85,
                "Urgent": 0.10,
                "Non-urgent": 0.05
              }
            };
          }
          // Urgent symptoms
          else if (['fever', 'abdominal pain', 'headache'].includes(primarySymptom) || severity >= 7) {
            response = {
              triage_category: "Urgent Care Recommended",
              confidence: 0.80,
              recommended_action: "Visit urgent care within 24 hours",
              probabilities: {
                "Emergency": 0.20,
                "Urgent": 0.60,
                "Non-urgent": 0.20
              }
            };
          }
          // Non-urgent symptoms
          else {
            response = {
              triage_category: "Non-urgent Care",
              confidence: 0.75,
              recommended_action: "Schedule a doctor's appointment",
              probabilities: {
                "Emergency": 0.05,
                "Urgent": 0.20,
                "Non-urgent": 0.75
              }
            };
          }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        $('#confidenceLevel').text((data.confidence * 100).toFixed(0) + '% confident');
        $('#confidenceBar').css('width', (data.confidence * 100) + '%');
        
        let probabilitiesHtml = '<div class="mt-3"><h5>Detailed Probabilities:</h5><ul class="list-group">';
        for (const [category, prob] of Object.entries(data.probabilities)) {
          probabilitiesHtml += `
            <li class="list-group-item d-flex justify-content-between align-items-center">
              ${category}
              <span class="badge badge-primary badge-pill">${(prob * 100).toFixed(0)}%</span>
            </li>`;
        }
        probabilitiesHtml += '</ul></div>';
        $('#probabilityDetails').html(probabilitiesHtml);
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });

              probabilities: {
                "Emergency": 0.05,

                "Non-urgent": 0.75
              }
            };

  <script>
    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

          // Enhanced triage logic based on symptoms
          setTimeout(() => {
            const primarySymptom = $('#primarySymptom').val().toLowerCase();
            const severity = parseInt($('#severity').val());
            const additionalSymptoms = $('#additionalSymptoms').val().toLowerCase();
            
            let response;
            
            // Emergency symptoms (life-threatening conditions)
            if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures', 'loss of consciousness', 'severe head injury'].includes(primarySymptom) || 
                severity >= 9 ||
                (additionalSymptoms.includes('difficulty breathing') && severity >= 7)) {
              response = {
                triage_category: "Emergency Care Needed",
                recommended_action: "Seek emergency care immediately",
                reasoning: "Your symptoms suggest a potentially life-threatening condition that requires immediate medical evaluation."
              };
            }
            // Urgent symptoms (should be seen within 24 hours)
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        
        // Clear previous probability display
        $('#confidenceLevel').remove();
        $('#confidenceBar').parent().remove();
        
        // Show reasoning in a clean format
        $('#probabilityDetails').html(`
          <div class="reasoning-box">
            <h5>Recommendation Details:</h5>
            <p>${data.reasoning}</p>
          </div>
        `);
        
        // Style based on urgency
        const resultBox = $('#resultContainer');
        resultBox.removeClass('emergency urgent home');
        if (data.triage_category.includes('Emergency')) {
          resultBox.addClass('emergency');
        } else if (data.triage_category.includes('Urgent')) {
          resultBox.addClass('urgent');
        } else {
          resultBox.addClass('home');
        }
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

          // Enhanced triage logic based on symptoms
          setTimeout(() => {
            const primarySymptom = $('#primarySymptom').val().toLowerCase();
            const severity = parseInt($('#severity').val());
            const additionalSymptoms = $('#additionalSymptoms').val().toLowerCase();
            
            let response;
            
            // Emergency symptoms (life-threatening conditions)
            if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures', 'loss of consciousness', 'severe head injury', 'sudden paralysis', 'severe allergic reaction'].includes(primarySymptom) || 
                severity >= 9 ||
                (additionalSymptoms.includes('difficulty breathing') && severity >= 7) ||
                (additionalSymptoms.includes('chest pain') && severity >= 7)) {
              response = {
                triage_category: "Emergency Care Needed",
                recommended_action: "Seek emergency care immediately",
                reasoning: "Your symptoms suggest a potentially life-threatening condition that requires immediate medical evaluation."
              };
            }
            // Urgent symptoms (should be seen within 24 hours)
            else if (['fever', 'abdominal pain', 'headache', 'dizziness', 'vomiting', 'diarrhea', 'rash with fever', 'eye pain', 'severe pain'].includes(primarySymptom) ||
                severity >= 7 ||
                (additionalSymptoms.includes('fever') && severity >= 5) ||
                (additionalSymptoms.includes('vomiting') && severity >= 5) ||
                (additionalSymptoms.includes('diarrhea') && ['Less than 1 day', '1-2 days'].includes($('#duration').val()))) {
              response = {
                triage_category: "Urgent Care Recommended",
                recommended_action: "Visit urgent care within 24 hours",
                reasoning: "Your symptoms require prompt medical attention but are not immediately life-threatening."
              };
            }
            // Home care (can wait for regular appointment)
            else {
              response = {
                triage_category: "Home Care Recommended",
                recommended_action: "Schedule a doctor's appointment if symptoms persist",
                reasoning: "Your symptoms appear to be mild and can likely be managed with self-care. Monitor your condition and seek medical attention if symptoms worsen or persist."
              };
            }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        
        // Clear previous probability display
        $('#confidenceLevel').remove();
        $('#confidenceBar').parent().remove();
        
        // Show reasoning in a clean format
        $('#probabilityDetails').html(`
          <div class="reasoning-box">
            <h5>Recommendation Details:</h5>
            <p>${data.reasoning}</p>
          </div>
        `);
        
        // Style based on urgency
        const resultBox = $('#resultContainer');
        resultBox.removeClass('emergency urgent home');
        if (data.triage_category.includes('Emergency')) {
          resultBox.addClass('emergency');
        } else if (data.triage_category.includes('Urgent')) {
          resultBox.addClass('urgent');
        } else {
          resultBox.addClass('home');
        }
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

  <!-- jQuery -->
  <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
  
  <script>
    $(document).ready(function() {
      // Initialize variables
      let currentStep = 1;
      const totalSteps = 4;
      let selectedSymptoms = [];
      
      // Update triage level as user fills form
      function updateTriageLevel() {
        let level = 0;
        
        // Age factor
        const age = parseInt($('#age').val()) || 0;
        if (age > 60 || age < 5) level += 10;
        
        // Primary symptom factor
        const primarySymptom = $('#primarySymptom').val().toLowerCase();
        if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures'].includes(primarySymptom)) {
          level += 30;
        } else if (primarySymptom) {
          level += 10;
        }
        
        // Additional symptoms factor
        level += selectedSymptoms.length * 5;
        
        // Severity factor
        const severity = parseInt($('#severity').val()) || 5;
        level += (severity - 1) * 3;
        
        // Cap at 100
        level = Math.min(100, Math.max(0, level));
        
        // Update indicator
        $('#triageProgress').css('width', level + '%');
        $('#triageLevelText').text(
          level < 30 ? 'Low' : 
          level < 60 ? 'Moderate' : 'High'
        );
        
        // Show/hide indicator
        if (level > 0) {
          $('#triageIndicator').fadeIn();
        } else {
          $('#triageIndicator').fadeOut();
        }
      }
      
      // Step navigation
      $('.next-step').click(function() {
        const nextStep = $(this).data('next');
        goToStep(nextStep);
      });
      
      $('.prev-step').click(function() {
        const prevStep = $(this).data('prev');
        goToStep(prevStep);
      });
      
      function goToStep(step) {
        // Validate current step before proceeding
        if (currentStep === 1 && !validateStep1()) return;
        if (currentStep === 2 && !validateStep2()) return;
        
        // Update UI
        $('.step-content').removeClass('active');
        $('.nav-step').removeClass('active');
        
        $(`#step${step}`).addClass('active');
        $(`.nav-step[data-step="${step}"]`).addClass('active');
        
        // Mark previous steps as completed
        for (let i = 1; i < step; i++) {
          $(`.nav-step[data-step="${i}"]`).addClass('completed');
        }
        
        currentStep = step;
        
        // Scroll to top of step
        $('html, body').animate({
          scrollTop: $('.assessment-container').offset().top - 20
        }, 300);
      }
      
      function validateStep1() {
        const age = $('#age').val();
        if (!age) {
          alert('Please enter your age');
          return false;
        }
        return true;
      }
      
      function validateStep2() {
        const primarySymptom = $('#primarySymptom').val();
        if (!primarySymptom) {
          alert('Please enter your primary symptom');
          return false;
        }
        return true;
      }
      
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        
        // Update selected symptoms array
        const symptomValue = $(this).data('value');
        const index = selectedSymptoms.indexOf(symptomValue);
        
        if (index === -1) {
          selectedSymptoms.push(symptomValue);
        } else {
          selectedSymptoms.splice(index, 1);
        }
        
        // Update hidden input
        $('#additionalSymptoms').val(selectedSymptoms.join(', '));
        
        // Update triage level
        updateTriageLevel();
      });
      
      // Body map interaction
      $('#bodyMap').click(function(e) {
        const rect = this.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        // Find which area was clicked (simplified for demo)
        let clickedArea = null;
        
        // Head area (simplified detection)
        if (y < rect.height * 0.25) {
          clickedArea = 'headArea';
        } 
        // Chest area
        else if (y < rect.height * 0.4) {
          clickedArea = 'chestArea';
        }
        // Abdomen area
        else if (y < rect.height * 0.6) {
          clickedArea = 'abdomenArea';
        }
        // Left arm
        else if (x < rect.width * 0.4 && y < rect.height * 0.6) {
          clickedArea = 'armLeftArea';
        }
        // Right arm
        else if (x > rect.width * 0.6 && y < rect.height * 0.6) {
          clickedArea = 'armRightArea';
        }
        // Left leg
        else if (x < rect.width * 0.5) {
          clickedArea = 'legLeftArea';
        }
        // Right leg
        else {
          clickedArea = 'legRightArea';
        }
        
        // Highlight selected area
        $('.body-area').removeClass('active');
        if (clickedArea) {
          $(`#${clickedArea}`).addClass('active');
          $('#painLocation').val($(`#${clickedArea}`).data('area'));
        } else {
          $('#painLocation').val('');
        }
      });
      
      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        // Show loading state
        $('button[type="submit"]').html('<i class="fas fa-spinner fa-spin"></i> Analyzing...').prop('disabled', true);
        
        // Simulate API call with timeout
        setTimeout(() => {
          // Process form data
          const formData = {
            age: $('#age').val(),
            gender: $('#gender').val(),
            primarySymptom: $('#primarySymptom').val(),
            additionalSymptoms: selectedSymptoms,
            duration: $('#duration').val(),
            severity: $('#severity').val(),
            painLocation: $('#painLocation').val(),
            allergies: $('#allergies').val()
          };
          
          // Generate triage result (simplified for demo)
          let result;
          const primarySymptom = formData.primarySymptom.toLowerCase();
          const severity = parseInt(formData.severity);
          if (primarySymptom === 'chest pain' && severity >= 7) {
            result = {
              triage_category: "Emergency Care Needed",
              recommended_action: "Seek emergency care immediately.",
              reasoning: "Your symptoms suggest a potentially life-threatening condition."
            };
          } else {
            result = {
              triage_category: "Routine Check",
              recommended_action: "Schedule a visit with your healthcare provider.",
              reasoning: "Your symptoms are manageable but should be evaluated."
            };
          }
          
          // Display results
          $('#triageResult').text(result.triage_category);
          $('#recommendedAction').text(result.recommended_action);
          $('#resultReasoning').text(result.reasoning);
          $('#resultContainer').fadeIn();
          
          // Reset button state
          $('button[type="submit"]').html('<i class="fas fa-stethoscope"></i> Get Recommendation').prop('disabled', false);
        }, 2000);
      });

      // Chatbot functionality
      $('#chatbotBtn').click(function() {
        $('#chatbotWindow').toggle();
      });

      $('#closeChatbot').click(function() {
        $('#chatbotWindow').hide();
      });

      $('#sendMessage').click(function() {
        const userMessage = $('#chatbotInput').val();
        if (userMessage) {
          $('#chatbotMessages').append(`
            <div class="chatbot-message user-message">${userMessage}</div>
          `);
          $('#chatbotInput').val('');
          
          // Simulate bot response
          setTimeout(() => {
            $('#chatbotMessages').append(`
              <div class="chatbot-message bot-message">I'm here to help! Please provide your symptoms.</div>
            `);
            $('#chatbotMessages').scrollTop($('#chatbotMessages')[0].scrollHeight);
          }, 1000);
        }
      });
    });
  </script>
